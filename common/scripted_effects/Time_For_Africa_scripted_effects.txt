Check_ROM_OIL = {
	ROM = {
	if = { limit = {check_variable = {ROM.ROM_LIBYA_OIL = 0 }} 
	swap_ideas = {
	remove_idea = ROM_LIBYA_OIL_1

	remove_idea = ROM_LIBYA_OIL_3
